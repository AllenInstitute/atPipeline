FROM ubuntu:18.04

#Prevent any questions about time zones
ENV TZ=US/Pacific
RUN ln -snf /usr/share/zoneinfo/$TZ /etc/localtime && echo $TZ > /etc/timezone

RUN apt-get update && apt-get install -y \
  xz-utils \
  build-essential \
  curl \
  cmake \ 
  cmake-gui \
  git \
  ninja-build \
  vim \
  xterm \
  vim-gnome \
  libboost-all-dev \
  clang

ENV CC=/usr/bin/clang
ENV CXX=/usr/bin/clang++
